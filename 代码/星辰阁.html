A. Madoka and Math Dad

Problem - A - Codeforces

为了使数字更大，有两种方法：一个是数字位数增加，另一个是将大数放在最开始一个位置。

但是在n固定的情况下，第一种方法的增加速度更快。所以为了尽可能获得更多的位数要选择小的数来组成新的数字。题意中还说不可以将相同的数放在一起，并且不可以出现0，所以可以选择1和2两个数字进行交替出现。每次一次循环，所给的n每次减小3，所以n%3剩余什么数就把哪个数放在第一位，才可以保证最后不会出现重复的情况。

  1. #include<iostream>
  2. #include<algorithm>
  3. using namespace std;
  4. int main(void)
  5. {
  6. 	int t;
  7. 	cin >> t;
  8. 	while (t--)
  9. 	{
  10. 		int n;
  11. 		cin >> n;
  12. 		if (n % 3==1)
  13. 		{
  14. 			while (n > 0)
  15. 			{
  16. 				if (n >= 1)
  17. 				{
  18. 					cout << 1;
  19. 					n -= 1;
  20. 				}
  21. 				if (n >= 2)
  22. 				{
  23. 					cout << 2;
  24. 					n -= 2;
  25. 				}
  26. 			}
  27. 			cout << endl;
  28. 		}
  29. 		else
  30. 		{
  31. 			while (n > 0)
  32. 			{
  33. 				if (n >= 2)
  34. 				{
  35. 					cout << 2;
  36. 					n -= 2;
  37. 				}
  38. 				if (n >= 1)
  39. 				{
  40. 					cout << 1;
  41. 					n -= 1;
  42. 				}
  43. 			}
  44. 			cout << endl;
  45. 		}
  46. 	}
  47. }

B. Madoka and the Elegant Gift

Problem - B - Codeforces

这个题的本质是找图里面是否存在完整的矩阵。如果不存在就输出NO，存在就输出YES。

10         01         11         11

11         11          01        10

凡是出现以上矩阵的皆为NO

在输入的时候，要注意题中给的每行是字符串，所以需要转换一下。

  1. #include<iostream>
  2. #include<algorithm>
  3. #include<string>
  4. #include<string.h>
  5. using namespace std;
  6. const int N = 2e3;
  7. int num[N][N];
  8. int n, m;
  9. bool search(int x, int y)
  10. {
  11. 	if (!num[x][y])
  12. 	{
  13. 		if (x > 0 && y > 0 && num[x][y - 1] && num[x - 1][y] && num[x - 1][y - 1])
  14. 			return true;
  15. 		if (y < m && x > 0 && num[x][y + 1] && num[x - 1][y + 1] && num[x - 1][y])
  16. 			return true;
  17. 		if (x < n && y > 0 && num[x][y - 1] && num[x + 1][y] && num[x + 1][y - 1])
  18. 			return true;
  19. 		if (x < n && y < m && num[x][y + 1] && num[x + 1][y] && num[x + 1][y + 1])
  20. 			return true;
  21. 	}
  22. 	return false;
  23. }
  24. int main(void)
  25. {
  26. 	int t;
  27. 	cin >> t;
  28. 	while (t--)
  29. 	{
  30. 		cin >> n >> m;
  31. 		memset(num, 0, sizeof(num));
  32. 		int flag = 1;
  33. 		for (int i = 0; i < n; i++)
  34. 		{
  35. 			string s;
  36. 			cin >> s;
  37. 			for (int j = 0; j < m; j++)
  38. 				num[i][j] = s[j] - '0';
  39. 		}
  40. 		for (int i = 0; i < n; i++)
  41. 		{
  42. 			for (int j = 0; j < m; j++)
  43. 				if (search(i, j))
  44. 				{
  45. 					flag = 0;
  46. 					break;
  47. 				}
  48. 			if (!flag)
  49. 				break;
  50. 		}
  51. 		if (flag)
  52. 			cout << "YES" << endl;
  53. 		else cout << "NO" << endl;
  54. 	}
  55. }

C. Madoka and Childish Pranks

Problem - 1647C - Codeforces

这个题的涂色方法是一个白色空格向左涂黑或者向下涂黑。显然，第一个（1，1）不能涂黑色。所以，为了解决黑色叠加的情况，可以从后进行涂色。一步一步的涂色。

  1. #include<iostream>
  2. #include<algorithm>
  3. #include<vector>
  4. #include<string>
  5. using namespace std;
  6. const int N = 200;
  7. typedef pair<pair<int, int>, pair<int, int>> pii;
  8. int num[N][N];
  9. int main(void)
  10. {
  11. 	int t;
  12. 	scanf("%d", &t);
  13. 	while (t--)
  14. 	{
  15. 		int n, m;
  16. 		scanf("%d %d", &n, &m);
  17. 		for (int i = 1; i <= n; i++)
  18. 		{
  19. 			string s;
  20. 			cin >> s;
  21. 			for (int j = 0; j < m; j++)
  22. 				num[i][j + 1] = s[j] - '0';
  23. 		}
  24. 		vector<pii> p;
  25. 		if (num[1][1])
  26. 		{
  27. 			cout << -1 << endl;
  28. 			continue;
  29. 		}
  30. 		int ans = 0;
  31. 		for (int i = n; i >= 1; i--)
  32. 			for (int j = m; j >= 1; j--)
  33. 			{
  34. 				if (num[i][j])
  35. 				{
  36. 					int flag = 0;
  37. 					if (i > 1)
  38. 					{
  39. 						p.push_back({ make_pair(i - 1,j),make_pair(i,j) });
  40. 						ans++;
  41. 						flag = 1;
  42. 					}
  43. 					if (j > 1 && !flag)
  44. 					{
  45. 						p.push_back({ make_pair(i,j - 1),make_pair(i,j) });
  46. 						ans++;
  47. 					}
  48. 				}
  49. 			}
  50. 			cout << ans << endl;
  51. 			for (int i = 0; i < ans; i++)
  52. 				cout << p[i].first.first << ' ' << p[i].first.second << ' ' << p[i].second.first << ' ' << p[i].second.second << endl;
  53. 	}
  54. }
